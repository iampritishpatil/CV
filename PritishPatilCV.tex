\cvsection{research experience}

\cvevent{Role of inhibitory interneurons in working memory}{Dr. Ofer Yizhar, Prof. Misha Tsodyks}{2017-ongoing}{Weizmann Institute of Science}
\begin{itemize}
\item Made new automated spatial working memory task for mice
\item Studied a theoretical model of working memory with various interneuron subtypes
\item Tested long duration inhibition of c-fos using stGtACR in the mouse mPFC
\end{itemize}

\divider



\cvevent{Testing st-GtACR in-vivo with fear conditioning}{Dr. Ofer Yizhar}{2017}{Weizmann Institute of Science}
\begin{itemize}
\item Injected mice with the st-GtACR in BLA
\item Used fear conditioning as an assay to quantify the effectiveness of silencing of st-GtACR to prevent the extinction of fear conditioning
\end{itemize}


\divider


\cvevent{Modeling and recording starburst amacrine cells}{Dr. Michal Rivlin}{2017}{Weizmann Institute of Science}
\begin{itemize}
\item Learned to patch clamp record from retinal neurons.
\item Worked to make detailed biophysically realistic models to simulate starburst amacrine cells to study their dendritic direction selectivity
\end{itemize}

\divider


\cvevent{Recovering synchronization of data}{Prof. Nachum Ulanovsky}{2016}{Weizmann Institute of Science}
\begin{itemize}
\item Analyzed collected data from bat location and electrophysiology to recover the lost synchronization between the data streams
\end{itemize}


\divider



\cvevent{Making a realistic model CA1 Pyramidal Neuron in MOOSE}{Prof. Upinder S Bhalla}{2015 – 2016}{NCBS Bangalore}
\begin{itemize}
\item Wrote an optimization routine to fit detailed biophysical models of neurons to experimental patch clamp data
\item Worked to extract features to be used in the optimization routine
\end{itemize}
